Documentation LQL
Sélection simple
SELECT name, details 
FROM users 
WHERE city == "Paris";
Sélection avec opérateur IN
SELECT name, details 
FROM users 
WHERE "Python" IN details.skills;
Comptage avec condition
COUNT FROM users 
WHERE salary > 45000;
Calcul de moyenne
AVG salary FROM users;
Accès aux champs imbriqués
SELECT name, details.skills 
FROM users 
WHERE details.department == "IT";
Opérateurs disponibles
  • == (égal)
  • != (différent)
  • > (supérieur)
  • < (inférieur)
  • >= (supérieur ou égal)
  • <= (inférieur ou égal)
  • IN (dans une liste)
Fonctions d'agrégation
MIN amount FROM orders;
MAX salary FROM users;
SUM amount FROM orders;
Conditions multiples
SELECT name, details 
FROM users 
WHERE city == "Paris" 
AND "Python" IN details.skills;
